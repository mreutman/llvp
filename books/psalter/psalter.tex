  \ifnum\value{page}>\value{headerstartpage}%
    \markboth{\textsc{\formatTextHeader cap. {\footnotesize #1}}}%
             {\textsc{\formatTextHeader cap. {\footnotesize #1}}}%

    \markboth{}{}% Clear headers if we haven't reached the page count yet
  \fi
}
